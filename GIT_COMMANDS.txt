# Git Commands to Push Changes to dashboard-apis Branch

# 1. Check current branch and status
git branch
git status

# 2. Create and switch to dashboard-apis branch (if it doesn't exist)
# If the branch already exists, just switch to it
git checkout dashboard-apis || git checkout -b dashboard-apis

# 3. Add only the changed files
git add BhuvisWebsite/client/src/pages/Dashboard/DashboardPage.js
git add BhuvisWebsite/client/src/pages/Dashboard/components/AirportDistanceWidget.js
git add BhuvisWebsite/client/src/pages/Dashboard/components/BusinessCategoryCard.js
git add BhuvisWebsite/DASHBOARD_CHANGES.md

# 4. Commit the changes with a descriptive message
git commit -m "Enhance dashboard with API integrations and UI improvements

- Added Airport Distance widget with API integration
- Fixed Business Category API integration
- Improved rent data handling with API
- Enhanced UI layout and responsiveness
- Added documentation of changes"

# 5. Push changes to the remote repository
# If this is a new branch:
git push -u origin dashboard-apis

# If the branch already exists:
git push

# 6. Verify the changes were pushed successfully
git log --oneline -5
